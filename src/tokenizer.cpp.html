44/*!
45 * \fn void tokenize(const char* str, size_t len)
46 * \param the string to tokenize
47 * \param the length of the string
48 */
49void Tokenizer::tokenize(const char* str, size_t len)
50{
51 m_chars = str;
52 m_len = len;
53 m_pos = 0;
54 m_start = 0;
55
56 if (state == State::LongComment)
57 readMultiLineComment();
58
59 while (!atEnd())
60 read();
61}

68 * vector.

71{
72 state = State::Default;
73 output.clear();

76void Tokenizer::read()
77{
78 consumeDiscardable();
79
80 if (atEnd())
81 return;
82
83 m_start = pos();
84 size_t p = m_start;
85
86 char c = readChar();
87 CharacterType ct = ctype(c);
88
89 switch (ct)
90 {
91 case Digit:
92 return readNumericLiteral();
93 case DoubleQuote:
94 return readStringLiteral();
95 case SingleQuote:
96 return readCharLiteral();
97 case Letter:
98 case Underscore:
99 return readIdentifier();
100 case LeftPar:
101 return write(TokenType::LeftPar);
102 case RightPar:
103 return write(TokenType::RightPar);
104 case LeftBrace:
105 return write(TokenType::LeftBrace);
106 case RightBrace:
107 return write(TokenType::RightBrace);
108 case LeftBracket:
109 return write(TokenType::LeftBracket);
110 case RightBracket:
111 return write(TokenType::RightBracket);
112 case Semicolon:
113 return write(TokenType::Semicolon);
114 case Colon:
115 return readColonOrColonColon();
116 case QuestionMark:
117 return write(TokenType::QuestionMark);
118 case Comma:
119 return write(TokenType::Comma);
120 case Dot:
121 return write(TokenType::Dot);
122 case Punctuator:
123 return readFromPunctuator(c);
124 default:
125 return write(TokenType::Invalid);
126 }
127}

323void Tokenizer::readNumericLiteral()
324{
325 if (atEnd()) {
326 if (charAt(m_start) == '0')
327 return write(TokenType::OctalLiteral);
328 else
329 return write(TokenType::IntegerLiteral);
330 }
331
332 char c = peekChar();
333
334 // Reading binary, octal or hexadecimal number
335 // eg. : 0b00110111
336 // 018
337 // 0xACDBE
338 if (charAt(m_start) == '0' && c != '.')
339 {
340 if (c == 'x') // hexadecimal
341 return readHexa();
342 else if (c == 'b') // binary
343 return readBinary();
344 else if (Tokenizer::isDigit(c))// octal
345 return readOctal();
346 else // it is zero
347 return write(TokenType::OctalLiteral);
348 }
349
350 return readDecimal();
351}
352
353void Tokenizer::readHexa()
354{
355 const char x = readChar();
356 assert(x == 'x');
357
358 if (atEnd()) // input ends with '0x' -> error
359 return write(TokenType::Invalid);
360
361 while (!atEnd() && Tokenizer::isHexa(peekChar()))
362 readChar();
363
364 if (pos() - m_start == 2) // e.g. 0x+
365 return write(TokenType::Invalid);
366
367 return write(TokenType::HexadecimalLiteral);
368}
369
370void Tokenizer::readOctal()
371{
372 while (!atEnd() && Tokenizer::isOctal(peekChar()))
373 readChar();
374
375 return write(TokenType::OctalLiteral);
376}
377
378void Tokenizer::readBinary()
379{
380 const char b = readChar();
381 assert(b == 'b');
382
383 if (atEnd()) // input ends with '0b' -> error
384 return write(TokenType::Invalid);
385
386 while (!atEnd() && Tokenizer::isBinary(peekChar()))
387 readChar();
388
389 return write(TokenType::BinaryLiteral);
390}

392void Tokenizer::readDecimal()
393{
394 // Reading decimal numbers
395 // eg. : 25
396 // 3.14
397 // 3.14f
398 // 100e100
399 // 6.02e23
400 // 6.67e-11
401
402 while (!atEnd() && Tokenizer::isDigit(peekChar()))
403 readChar();
404
405 if (atEnd())
406 return write(TokenType::IntegerLiteral);
407
408 bool is_decimal = false;
409
410 if (peekChar() == '.')
411 {
412 readChar();
413 is_decimal = true;
414
415 while (!atEnd() && Tokenizer::isDigit(peekChar()))
416 readChar();
417
418 if (atEnd())
419 return write(TokenType::DecimalLiteral);
420 }
421
422 if (peekChar() == 'e')
423 {
424 readChar();
425 is_decimal = true;
426
427 if (atEnd())
428 return write(TokenType::Invalid);
429
430 if (peekChar() == '+' || peekChar() == '-')
431 {
432 readChar();
433 if (atEnd())
434 return write(TokenType::Invalid);
435 }
436
437 while (!atEnd() && Tokenizer::isDigit(peekChar()))
438 readChar();
439
440 if (atEnd())
441 return write(TokenType::DecimalLiteral);
442 }
443
444
445 if (peekChar() == 'f') // eg. 125.f
446 {
447 readChar();
448 is_decimal = true;
449 }
450 else
451 {
452 if (tryReadLiteralSuffix())
453 return write(TokenType::UserDefinedLiteral);
454 }
455
456 return write(is_decimal ? TokenType::DecimalLiteral : TokenType::IntegerLiteral);
457}
458
459bool Tokenizer::tryReadLiteralSuffix()
460{
461 auto cpos = pos();
462
463 if (!this->atEnd() && (Tokenizer::isLetter(peekChar()) || peekChar() == '_'))
464 readChar();
465 else
466 return false;
467
468 while (!this->atEnd() && (Tokenizer::isLetter(peekChar()) || Tokenizer::isDigit(peekChar()) || peekChar() == '_'))
469 readChar();
470
471 const bool read = (cpos != pos());
472 return read;
473}

475void Tokenizer::readPreprocessor()
476{
477 consumeDiscardable();
478
479 if (atEnd() || !isIdentifier(peekChar()))
480 return write(TokenType::Invalid);
481
482 while (!atEnd() && isIdentifierOrDigit(peekChar()))
483 readChar();
484
485 write(TokenType::Preproc);
486
487 if (this->output.back().text() == "#include")
488 {
489 consumeDiscardable();
490 m_start = pos();
491
492 if (atEnd() || (peekChar() != '<' && peekChar() != '"'))
493 return;
494
495 char c = readChar();
496 c = c == '<' ? '>' : '"';
497
498 while (!atEnd() && peekChar() != c)
499 readChar();
500
501 if (atEnd())
502 return write(TokenType::Invalid);
503
504 readChar();
505
506 return write(TokenType::Include);
507 }
508}
509
510void Tokenizer::readIdentifier()
511{
512 while (!this->atEnd() && (Tokenizer::isLetter(peekChar()) || Tokenizer::isDigit(peekChar()) || peekChar() == '_'))
513 readChar();
514
515 return write(identifierType(m_start, pos()));
516}

673void Tokenizer::readStringLiteral()
674{
675 while (!atEnd() && peekChar() != '"')
676 {
677 if (peekChar() == '\\')
678 {
679 readChar();
680 if (!atEnd())
681 readChar();
682 }
683 else if (peekChar() == '\n')
684 {
685 return write(TokenType::Invalid);
686 }
687 else
688 {
689 readChar();
690 }
691 }
692
693 if(atEnd())
694 return write(TokenType::Invalid);
695
696 assert(peekChar() == '"');
697 readChar();
698
699 if (tryReadLiteralSuffix())
700 return write(TokenType::UserDefinedLiteral);
701
702 return write(TokenType::StringLiteral);
703}
704
705void Tokenizer::readCharLiteral()
706{
707 if(atEnd())
708 return write(TokenType::Invalid);
709
710 readChar();
711
712 if (atEnd())
713 return write(TokenType::Invalid);
714
715 if(ctype(readChar()) != SingleQuote)
716 return write(TokenType::Invalid);
717
718 return write(TokenType::StringLiteral);
719}

721void Tokenizer::readFromPunctuator(char p)
722{
723 if (p == '/')
724 {
725 if (atEnd())
726 return write(TokenType::Div);
727 if (peekChar() == '/')
728 return readSingleLineComment();
729 else if (peekChar() == '*')
730 return readMultiLineComment();
731 else
732 return readOperator();
733 }
734 else if (p == '#')
735 {
736 return readPreprocessor();
737 }
738
739 return readOperator();
740}

743void Tokenizer::readColonOrColonColon()
744{
745 if (atEnd())
746 return write(TokenType::Colon);
747
748 if (peekChar() == ':')
749 {
750 readChar();
751 return write(TokenType::ScopeResolution);
752 }
753
754 return write(TokenType::Colon);
755}

837void Tokenizer::readOperator()
838{
839 TokenType op = getOperator(m_start, pos());
840
841 if (op == TokenType::Invalid)
842 return write(TokenType::Invalid);
843
844 while (!atEnd())
845 {
846 size_t p = pos();
847 readChar();
848 TokenType candidate = getOperator(m_start, pos());
849
850 if (candidate == TokenType::Invalid)
851 {
852 --m_pos;
853 break;
854 }
855 else
856 {
857 op = candidate;
858 }
859 }
860
861 return write(op);
862}

864void Tokenizer::readSingleLineComment()
865{
866 readChar(); // reads the second '/'
867
868 while (!atEnd() && peekChar() != '\n')
869 readChar();
870
871 return write(TokenType::SingleLineComment);
872}

874void Tokenizer::createLongComment()
875{
876 state = State::LongComment;
877 return write(TokenType::MultiLineComment);
878}
879
880void Tokenizer::readMultiLineComment()
881{
882 if(state == State::Default)
883 readChar(); // reads the '*' after opening '/'
884
885 do {
886 while (!atEnd() && peekChar() != '*')
887 readChar();
888
889 if (atEnd())
890 return createLongComment();
891
892 assert(peekChar() == '*');
893 readChar(); // reads the '*'
894
895 if (atEnd())
896 return createLongComment();
897
898 } while (peekChar() != '/');
899
900 readChar(); // reads the closing '/'
901 state = State::Default;
