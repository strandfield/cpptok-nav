232/*!
233 * \class Token
234 * \brief represents a token
235 *
236 * Each token has a \c{type()} and a \c{text()}.
237 *
238 * Note that the token's text is stored as a string_view.
239 * This makes tokens cheap to copy but requires that the string
240 * that was used to create the token outlives the token.
241 */
242
243class Token
244{
245private:
246 TokenType type_;
247 string_view str_;
248
249public:
250 Token();
251 Token(const Token & ) = default;
252 ~Token() = default;
253
254 Token(TokenType t, string_view str);
255
256 bool isValid() const;
257
258 TokenType type() const;
259 string_view text() const;
260
261 std::string to_string() const;
262
263 bool isOperator() const;
264 bool isIdentifier() const;
265 bool isKeyword() const;
266 bool isLiteral() const;
267 bool isComment() const;
268
269 bool operator==(const Token& other) const { return type_ == other.type_ && other.str_ == str_; }
270 bool operator!=(const Token & other) const { return !operator==(other); }
271 bool operator==(TokenType tok) const { return type_ == tok; }
272 bool operator!=(TokenType tok) const { return !operator==(tok); }
273
274 Token & operator=(const Token&) = default;
275};
